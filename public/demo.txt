app.post('/api/developers', async (req, res) => {
  const {
    name,
    role,
    experience,
    skills,
    bio,
    rating,
    followers,
    following
  } = req.body;

  const result = await pool.query(
    `INSERT INTO developers 
      (name, role, experience, skills, bio, rating, followers, following)
      VALUES ($1, $2, $3, $4, $5, $6, $7, $8)
      RETURNING *`,
    [name, role, experience, skills, bio, rating, followers, following]
  );

  res.status(201).json(result.rows[0]);
  console.log(result.rows[0]);
});

app.get('/api/developers/:id', async (req, res) => {
  const { id } = req.params;
  const devRes = await pool.query('SELECT * FROM developers WHERE id = $1', [id]);

  const dev = devRes.rows[0];

  const projRes = await pool.query('SELECT * FROM projects WHERE developer_id = $1', [id]);
  dev.projects = projRes.rows;

import React from 'react';
import { NavigationContainer } from '@react-navigation/native';
import { createNativeStackNavigator } from '@react-navigation/native-stack';
import HomeScreen from '../screens/HomeScreen';
import DeveloperDetailScreen from '../screens/DeveloperDetailScreen';
import AddDeveloperScreen from '../screens/AddDeveloperScreen';

export type Developer = {
    id: string;
    name: string;
    role: string;
    experience: string;
    skills: string[];
    bio: string;
};

export type RootStackParamList = {
    Home: undefined;
    DeveloperDetail: { developer: Developer };
    AddDeveloper: undefined;
};

const Stack = createNativeStackNavigator<RootStackParamList>();

const AppNavigator = () => {
    return (
        <NavigationContainer>
            <Stack.Navigator
                initialRouteName='Home'
                screenOptions={{ headerShown: false }}>

                <Stack.Screen name="Home" component={HomeScreen} />
                <Stack.Screen name="DeveloperDetail" component={DeveloperDetailScreen} />
                <Stack.Screen name="AddDeveloper" component={AddDeveloperScreen} />

            </Stack.Navigator>
        </NavigationContainer>
    );
};

export default AppNavigator;

type AddDeveloperScreenNavigationProp = NativeStackNavigationProp<RootStackParamList, 'AddDeveloper'>;

type Props = {
    navigation: AddDeveloperScreenNavigationProp;
};

const AddDeveloperScreen: React.FC<Props> = ({ navigation }) => {
    const isDarkMode = useColorScheme() === 'dark';
    const [loading, setLoading] = useState(false);

    const [name, setName] = useState('');
    const [role, setRole] = useState('');
    const [experience, setExperience] = useState('');
    const [skills, setSkills] = useState('');
    const [bio, setBio] = useState('');

    const handleAddDeveloper = async () => {
        if (!name || !role || !experience || !skills || !bio) {
            Alert.alert('Error', 'Please fill in all fields');
            return;
        }

        setLoading(true);
        const API_URL = 'http://192.168.1.25:3000';

        const newDev = {
            name,
            role,
            experience,
            skills: skills.split(',').map(s => s.trim()),
            bio,
            rating: 0,
            followers: 0,
            following: 0
        };

        try {
            const response = await fetch(`${API_URL}/api/developers`, {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json',
                },
                body: JSON.stringify(newDev),
            });

            if (response.ok) {
                Alert.alert('Success', 'Developer added successfully!', [
                    { text: 'OK', onPress: () => navigation.navigate('Home') }
                ]);
            } else {
                const errorData = await response.json();
                Alert.alert('Error', errorData.error || 'Failed to add developer');
            }
        } catch (error) {
            console.error('Add Developer Error:', error);
            Alert.alert('Error', 'Could not connect to server');
        } finally {
            setLoading(false);

                    <View style={styles.inputGroup}>
                        <Text style={styles.label}>Professional Bio</Text>
                        <TextInput
                            style={[styles.input, styles.textArea]}
                            placeholder="Tell us about your expertise..."
                            placeholderTextColor="#888"
                            value={bio}
                            onChangeText={setBio}
                            multiline
                        />
                    </View>

                    <TouchableOpacity
                        style={styles.submitBtn}
                        onPress={handleAddDeveloper}
                        disabled={loading}
                    >
                        {loading ? (
                            <ActivityIndicator color="#FFF" />
                        ) : (
                            <Text style={styles.submitBtnText}>Save Developer</Text>
                        )}
                    </TouchableOpacity>

                    <TouchableOpacity style={styles.backLink} onPress={() => navigation.goBack()}>
                        <Text style={styles.backLinkText}>Cancel</Text>
                    </TouchableOpacity>
                </View>
            </ScrollView>
        </SafeAreaView>
    );
};

export default AddDeveloperScreen;
